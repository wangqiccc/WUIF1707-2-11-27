1.所有商品平均价格

SELECT avg(merprice) as avgnum FROM merchan;

2.某一类型商品平均价格

SELECT avg(merprice) as avgnum FROM merchan WHERE tid=2;

3.查询某个商品的存货价值

SELECT merprice*invent as value FROM merchan WHERE mername="芒果";

4.查询某个订单的金额

SELECT merprice*gnumber FROM merchan,goodsorder WHERE merchan.id=goodsorder.goodid and goodsorder.goodid="7";

5.价格最低三个产品

SELECT * FROM merchan ORDER BY merprice LIMIT 0,3;

6.最新添加的一个产品

SELECT * FROM merchan ORDER BY id DESC LIMIT 0,1;

7.商品所有信息和类型

SELECT * FROM merchan,type WHERE merchan.tid=type.id;

8.销量最高三个产品

SELECT * FROM goodsorder ORDER BY gnumber DESC LIMIT 0,3;
