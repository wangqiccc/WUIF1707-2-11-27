1. 查询所有商品的平均价格
select avg(gprice) from goods;
2. 查询某一类型商品的平均价格
select avg(gprice) from goods where tid=1;
3. 查询某个商品的存货价值
select gprice*gsurplus from goods where id=6;
4. 查询某个订单的金额
select goods.gprice*orders.number from orders,goods where goods.id=orders.gid and orders.id=1;
5. 查询价格最低的三个商品
select * from goods order by gprice limit 0,3;
6. 查询最新添加的一个商品
select * from goods order by id desc limit 0,1;
7. 查询商品的所有信息以及它的类型
select goods.*,type.tname from goods,type where goods.tid=type.id;
8. 查询销量最高的三个商品
select goods.gname,orders.number from goods,orders where goods.id=orders.gid order by orders.number desc limit 0,3;
